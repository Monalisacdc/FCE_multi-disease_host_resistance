starts <- dat_env_ts %>%
  filter(between(time, dae_min, dae_max)) %>%
  group_by(env, geno) %>%
  summarise(
    t_pos = {
      v <- time[Tan_spot > 0]
      if (length(v) == 0) NA_real_ else min(v)},

1. Helper Function: Check and Align Matrices

This function receives several distance matrices and keeps only the cultivars that are present in all of them.

Code 
align_distance_matrices <- function(...) {
  mats <- list(...)
  
  # Intersection of cultivars
  common_ids <- Reduce(intersect, lapply(mats, rownames))
  
  mats2 <- lapply(mats, function(M) {
    M[common_ids, common_ids, drop = FALSE]
  })
  
  return(mats2)
}

2. Function: Distance to Resistant Group

For each cultivar, calculate the average distance to the cultivars in the resistant cluster.

Code 
distance_to_resistant_group <- function(D, clusters, resistant_cluster) {
  
  # Ensure alignment between vector and matrix names
  ids <- intersect(rownames(D), names(clusters))
  D <- D[ids, ids, drop = FALSE]
  clusters <- clusters[ids]
  
  resistant_ids <- names(clusters)[clusters == resistant_cluster]
  
  if(length(resistant_ids) < 1) {
    stop("No cultivar found in the resistant cluster.")
  }
  
  # Average distance to the resistant group
  dist_mean <- sapply(ids, function(g) {
    mean(D[g, resistant_ids], na.rm = TRUE)
  })
  
  # Optional: exclude self-distance if g is also in the resistant group
  dist_mean2 <- sapply(ids, function(g) {
    ref <- resistant_ids
    if(g %in% ref && length(ref) > 1) ref <- setdiff(ref, g)
    if(length(ref) == 0) return(NA_real_)
    mean(D[g, ref], na.rm = TRUE)
  })
  
  tibble(
    genotype = ids,
    dist_to_resistant = dist_mean2
  )
}

```{r}
align_distance_matrices <- function(...) {
  mats <- list(...)
  
  # Intersection of cultivars
  common_ids <- Reduce(intersect, lapply(mats, rownames))
  
  mats2 <- lapply(mats, function(M) {
    M[common_ids, common_ids, drop = FALSE]
  })
  
  return(mats2)
}
```

## 2. Function: Distance to Resistant Group

For each cultivar, calculate the average distance to the cultivars in the resistant cluster.

```{r}
distance_to_resistant_group <- function(D, clusters, resistant_cluster) {
  
  # Ensure alignment between vector and matrix names
  ids <- intersect(rownames(D), names(clusters))
  D <- D[ids, ids, drop = FALSE]
  clusters <- clusters[ids]
  
  resistant_ids <- names(clusters)[clusters == resistant_cluster]
  
  if(length(resistant_ids) < 1) {
    stop("No cultivar found in the resistant cluster.")
  }
  
  # Average distance to the resistant group
  dist_mean <- sapply(ids, function(g) {
    mean(D[g, resistant_ids], na.rm = TRUE)
  })
  
  # Optional: exclude self-distance if g is also in the resistant group
  dist_mean2 <- sapply(ids, function(g) {
    ref <- resistant_ids
    if(g %in% ref && length(ref) > 1) ref <- setdiff(ref, g)
    if(length(ref) == 0) return(NA_real_)
    mean(D[g, ref], na.rm = TRUE)
  })
  
  tibble(
    genotype = ids,
    dist_to_resistant = dist_mean2
  )
}
```

```{r}
edge_gradient <- function(p1, p2, col1, col2, n = 200) {
  t_vals <- seq(0, 1, length.out = n)
  
  x_vals <- (1 - t_vals) * p1[1] + t_vals * p2[1]
  y_vals <- (1 - t_vals) * p1[2] + t_vals * p2[2]
  
  tibble::tibble(
    x = x_vals[-n],
    y = y_vals[-n],
    xend = x_vals[-1],
    yend = y_vals[-1],
    col = grDevices::colorRampPalette(c(col1, col2))(n - 1)
  )
}
```
